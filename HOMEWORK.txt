3. 	$a = 5;
	Переменной $a присваевается значение 5 с числовым типом данных;

	$b = '05';
	Переменной $b присваевается значение 05 с текстовым типом данных;

	var_dump($a == $b);
	True потому что php смотрит что в переменной $a записано число и в переменной $b записано число, при математической операции == сравниваются только значения переменных, а там числа, поэтому true

	var_dump((int)'012345');
	Строчный тип данных '012345' явно приводится к числовому типу данных, поэтому строка 012345 становится числом 12345
	
	var_dump((float)123.0 === (int)123.0);
	Значения 123.0 и 123.0 явно приводятся к типам данных, к float - число с плавоющей точкой, int - целое число, при матемотической операции === выполняется сравнение по типу данных и значениях, значения одинаковые, а типы данных разные, поэтому false

	var_dump((int)0 === (int)'hello, world');
	Значения 0 и 'hello, world' явно приводятся к одному типу данных int, 0 это 0, 'hello, world' или любой текст это 0, при матемотической операции === выполняется сравнение по типу данных и значениям, они равны, поэтому true
